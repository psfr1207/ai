2단계. 편집창에 아래 코드 전체를 붙여넣기
// ===================================================== // API 키·Sheet ID는 코드에 직접 쓰지 않습니다. // GAS 편집기 → 프로젝트 설정 → 스크립트 속성에서 // 아래 두 키를 추가하세요: // ANTHROPIC_API_KEY → sk-ant-api03-... // SHEET_ID → 스프레드시트 ID // ===================================================== const SHEET_NAME = '대화로그'; // ===================================================== // doPost: Claude API 프록시 + 로그 저장 통합 // ===================================================== function doPost(e) { try { const props = PropertiesService.getScriptProperties(); const API_KEY = props.getProperty('ANTHROPIC_API_KEY'); const SHEET_ID = props.getProperty('SHEET_ID'); if (!API_KEY) throw new Error('API 키가 설정되지 않았습니다. setSecrets()를 먼저 실행하세요.'); const raw = e.parameter.payload || e.parameter.data || (e.postData && e.postData.contents) || '{}'; const body = JSON.parse(raw); const action = e.parameter.action || body.action || 'chat'; // [1] chat: Claude API 프록시 if (action === 'chat') { const apiResp = UrlFetchApp.fetch( 'https://api.anthropic.com/v1/messages', { method: 'post', contentType: 'application/json', headers: { 'x-api-key': API_KEY, 'anthropic-version': '2023-06-01' }, payload: JSON.stringify({ model: body.model || 'claude-sonnet-4-20250514', max_tokens: body.max_tokens || 600, system: body.system || '', messages: body.messages || [] }), muteHttpExceptions: true } ); const result = JSON.parse(apiResp.getContentText()); return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // [2] logs: 대화 로그 저장 if (action === 'logs') { const ss = SpreadsheetApp.openById(SHEET_ID); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['세션ID','타임스탬프','학생ID','학년', '차시번호','차시명','유형','단계번호','단계명','발화자','내용']); } const rows = body.logs || []; rows.forEach(r => sheet.appendRow([ r.session_id, r.timestamp, r.student_id, r.grade, r.lesson_id, r.lesson_title, r.mode_label, r.step, r.step_label, (r.role === 'assistant' ? 'AI' : r.role === 'user' ? '학생' : r.role), r.content ])); return ContentService .createTextOutput(JSON.stringify({status:'ok', count: rows.length})) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput(JSON.stringify({ok:false, error:'알 수 없는 action: ' + action})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false, error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { var props = PropertiesService.getScriptProperties(); var secret = props.getProperty('READ_KEY'); // 설정 시에만 잠금 적용 if (secret && String((e && e.parameter && e.parameter.key) || '') !== secret) { return ContentService.createTextOutput(JSON.stringify({ok:false, error:'unauthorized'})).setMimeType(ContentService.MimeType.JSON); } var ss = SpreadsheetApp.openById(props.getProperty('SHEET_ID')); var sh = ss.getSheetByName(SHEET_NAME); if (!sh) return ContentService.createTextOutput('[]').setMimeType(ContentService.MimeType.JSON); var values = sh.getDataRange().getValues(); var head = values.shift() || []; var rows = values.map(function(r){ var o={}; head.forEach(function(h,i){ o[h]=r[i]; }); return o; }); return ContentService.createTextOutput(JSON.stringify(rows)).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ok:false, error: err.toString()})).setMimeType(ContentService.MimeType.JSON); } }
3단계. 왼쪽 메뉴 ⚙️ 프로젝트 설정 → 하단 스크립트 속성 → 속성 추가
   • 속성: ANTHROPIC_API_KEY / 값: API 키 (sk-ant-api03-...)
   • 속성: SHEET_ID / 값: 스프레드시트 URL의 긴 ID
